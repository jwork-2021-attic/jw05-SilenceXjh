    private String fontFilename;

    public String getFontFilename() {
        return fontFilename;
    }

    private int width;

    public int getWidth() {
        return width;
    }

    private int height;

    public int getHeight() {
        return height;

    public AsciiFont(String filename, int width, int height) {
        this.fontFilename = filename;
